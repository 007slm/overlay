Overlay - 全局遮罩Mask组件示例

默认遮罩层

seajs.use('../src/mask', function(mask) {
    document.getElementById('a').onclick = function() {
        mask.show();
    };
});

修改遮罩层(绿色、透明度0.3)

seajs.use('../src/mask', function(mask) {
    document.getElementById('b').onclick = function() {
        mask.set('backgroundColor', 'green').set('opacity', '0.3').show();
    };
});
